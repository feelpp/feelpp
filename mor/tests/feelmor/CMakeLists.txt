foreach( geo test_deim.geo test_geim.geo parameterspace.json )

  add_custom_command(
        TARGET copy-${geo} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
                ${CMAKE_CURRENT_SOURCE_DIR}/${geo}

endforeach()
